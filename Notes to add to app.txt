package com.fairware.fairware_lift.wear

import android.Manifest
import android.app.Activity
import android.os.Bundle
import android.util.Log
import android.widget.TextView
import androidx.activity.result.contract.ActivityResultContracts
import androidx.health.services.client.HealthServices
import androidx.health.services.client.MeasureCallback
import androidx.health.services.client.data.Availability
import androidx.health.services.client.data.DataType
import androidx.health.services.client.data.MeasureData
import kotlinx.coroutines.*

class MainActivity : Activity() {

    private lateinit var textView: TextView
    private val healthServicesClient by lazy { HealthServices.getClient(this) }
    private val measureClient by lazy { healthServicesClient.measureClient }
    private val mainScope = CoroutineScope(Dispatchers.Main + SupervisorJob())

    private val permissionLauncher = registerForActivityResult(
        ActivityResultContracts.RequestPermission()
    ) { isGranted ->
        if (isGranted) {
            Log.d("WearHR", "BODY_SENSORS permission granted.")
            startHeartRateMonitoring()
        } else {
            Log.d("WearHR", "BODY_SENSORS permission denied.")
            textView.text = "Permission Denied"
        }
    }

    private val measureCallback = object : MeasureCallback {
        override fun onAvailabilityChanged(dataType: DataType, availability: Availability) {
            Log.d("WearHR", "Availability changed for $dataType: $availability")
            if (availability is Availability.Unavailable) {
                textView.text = "HR Sensor Unavailable"
            }
        }

        override fun onData(data: List<MeasureData>) {
            val heartRateBpm = data.firstOrNull()?.value?.asDouble()
            if (heartRateBpm != null) {
                textView.text = "HR: ${heartRateBpm.toInt()} bpm"
            }
        }
    }

    override fun onCreate(savedInstanceState: Bundle?) {
        super.onCreate(savedInstanceState)
        
        textView = TextView(this).apply {
            text = "Initializing..."
            textAlignment = TextView.TEXT_ALIGNMENT_CENTER
            textSize = 24f // Make text larger for watch face
        }
        
        setContentView(textView)
    }

    override fun onStart() {
        super.onStart()
        Log.d("WearHR", "onStart: Requesting permission.")
        permissionLauncher.launch(Manifest.permission.BODY_SENSORS)
    }

    override fun onStop() {
        super.onStop()
        Log.d("WearHR", "onStop: Unregistering HR listener.")
        mainScope.launch {
            try {
                measureClient.unregisterMeasureCallback(DataType.HEART_RATE_BPM, measureCallback)
            } catch (e: Exception) {
                Log.e("WearHR", "Error unregistering callback", e)
            }
        }
    }

    private fun startHeartRateMonitoring() {
        mainScope.launch {
            try {
                val capabilities = measureClient.getCapabilities()
                if (DataType.HEART_RATE_BPM in capabilities.supportedDataTypes) {
                    Log.d("WearHR", "HEART_RATE_BPM is supported. Registering callback.")
                    textView.text = "Starting HR..."
                    measureClient.registerMeasureCallback(DataType.HEART_RATE_BPM, measureCallback)
                } else {
                    Log.d("WearHR", "HEART_RATE_BPM is not supported on this device.")
                    textView.text = "No HR Sensor"
                }
            } catch (e: Exception) {
                Log.e("WearHR", "Error getting capabilities or registering callback", e)
                textView.text = "Error"
            }
        }
    }
}
